<script>

let current = 0;

let clicks = 0;

let rotations = Array(8).fill(0);


const grid = document.getElementById("grid");


for (let i = 0; i < 8; i++) {

  const peg = document.createElement("div");

  peg.className = "peg";


  const bar = document.createElement("div");

  bar.className = "bar";


  peg.appendChild(bar);

  grid.appendChild(peg);


  peg.onclick = () => {

    if (i !== current) return;


    rotations[i] += 90;

    bar.style.transform = `rotate(${rotations[i]}deg)`;


    clicks++;


    if (rotations[i] >= 360) {

      rotations[i] = 0;

      bar.style.transform = "rotate(0deg)";

      current = (current + 1) % 8;

      updateActive();

    }

  };

}


function updateActive() {

  document.querySelectorAll(".peg").forEach((p, i) => {

    p.classList.toggle("active", i === current);

  });

}


updateActive();


// TIMER

let time = 180;

const timerEl = document.getElementById("timer");


setInterval(() => {

  time--;

  let m = Math.floor(time / 60);

  let s = time % 60;

  timerEl.textContent = `${m}:${s < 10 ? "0" : ""}${s}`;


  if (time <= 0) {

    timerEl.textContent = "Finished";

  }

}, 1000);

</script>
